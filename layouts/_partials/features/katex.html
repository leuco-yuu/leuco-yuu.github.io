{{- $globalKatex := site.Params.katex -}} {{- $globalEnabled := false -}} {{- $globalDelimiters := slice -}} {{- if $globalKatex -}} {{- if eq (printf "%T" $globalKatex) "bool" -}} {{- $globalEnabled = $globalKatex -}} {{- else -}} {{- $globalEnabled = $globalKatex.enabled | default false -}} {{- $globalDelimiters = $globalKatex.delimiters -}} {{- end -}} {{- end -}} {{- $pageKatex := .Params.katex -}} {{- $pageEnabled := false -}} {{- $pageDelimiters := slice -}} {{- if $pageKatex -}} {{- if eq (printf "%T" $pageKatex) "bool" -}} {{- $pageEnabled = $pageKatex -}} {{- else -}} {{- $pageEnabled = $pageKatex.enabled | default false -}} {{- $pageDelimiters = $pageKatex.delimiters -}} {{- end -}} {{- end -}} {{- $featureEnabled := or $globalEnabled $pageEnabled -}} {{- $contentToScan := .Content -}} {{- /* 也扫描分页博文的子 .md 文件，使 KaTeX 检测能覆盖多文件文章 */ -}} {{- $sections := .Params.sections -}} {{- if $sections -}} {{- range $sections -}} {{- $filePath := path.Join "content" $.File.Dir . -}} {{- $contentToScan = printf "%s\n%s" $contentToScan (os.ReadFile $filePath) -}} {{- end -}} {{- else -}} {{- range .Resources.Match "*.md" -}} {{- if ne .Name "index.md" -}} {{- $filePath := path.Join "content" $.File.Dir .Name -}} {{- $contentToScan = printf "%s\n%s" $contentToScan (os.ReadFile $filePath) -}} {{- end -}} {{- end -}} {{- end -}} {{- $hasBlockMath := gt (len (findRE `\$\$` $contentToScan)) 0 -}} {{- $hasParenMath := gt (len (findRE `\\\(` $contentToScan)) 0 -}} {{- $hasBracketMath := gt (len (findRE `\\\[` $contentToScan)) 0 -}} {{- $hasInlineMath := gt (len (findRE `(^|[^$])\$[^$\n]+\$` $contentToScan)) 0 -}} {{- $shouldLoad := and $featureEnabled (or $hasBlockMath $hasParenMath $hasBracketMath $hasInlineMath) -}} {{- if $shouldLoad -}} {{- end -}}
